dtparam=spi=on
dtparam=i2c_arm=on

# Display — start with fbtft (docs/PLAN.md §5.2). Switch to the
# panel-mipi-dbi block below once the DRM path is validated.
#
# width/height are the panel's native (portrait) raster; rotate=90 turns it
# into the 320x240 landscape framebuffer the UI expects (docs/PLAN.md §5, §8
# — click-wheel iPods had landscape screens). If "rPod" renders upside-down
# or sideways once deployed, try rotate=270 instead.
dtoverlay=fbtft,spi0-0,st7789v,width=240,height=320,rotate=90,reset_pin=27,dc_pin=24,led_pin=13,speed=62500000
